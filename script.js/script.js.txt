const apiKey = 'YOUR_API_KEY'; // Replace with your API key
const watchButton = document.getElementById('watchButton');
const pairInput = document.getElementById('pair');
const slider = document.getElementById('slider');

watchButton.addEventListener('click', () => {
    const pair = pairInput.value.trim().toUpperCase();
    if (pair) {
        fetchData(pair);
    } else {
        alert("Please enter a valid pair.");
    }
});

async function fetchData(pair) {
    try {
        // Example: Fetching crypto data (adjust URL for Forex)
        const response = await fetch(`https://api.binance.com/api/v3/trades?symbol=${pair}&limit=10`);
        const data = await response.json();
        displayData(data);
    } catch (error) {
        console.error("Error fetching data:", error);
    }
}

function displayData(data) {
    slider.innerHTML = ''; // Clear previous data
    data.forEach(trade => {
        const tradeElement = document.createElement('div');
        tradeElement.textContent = `Price: ${trade.price}, Quantity: ${trade.qty}, Time: ${new Date(trade.time).toLocaleString()}`;
        slider.appendChild(tradeElement);
    });
}
